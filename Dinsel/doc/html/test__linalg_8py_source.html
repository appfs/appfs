245 def _stride_comb_iter(x):
246  """
247  Generate cartesian product of strides for all axes
248  """
249 
250  if not isinstance(x, np.ndarray):
251  yield x, "nop"
252  return
253 
254  stride_set = [(1,)] * x.ndim
255  stride_set[-1] = (1, 3, -4)
256  if x.ndim > 1:
257  stride_set[-2] = (1, 3, -4)
258  if x.ndim > 2:
259  stride_set[-3] = (1, -4)
260 
261  for repeats in itertools.product(*tuple(stride_set)):
262  new_shape = [abs(a * b) for a, b in zip(x.shape, repeats)]
263  slices = tuple([slice(None, None, repeat) for repeat in repeats])
264 
265  # new array with different strides, but same data
266  xi = np.empty(new_shape, dtype=x.dtype)
267  xi.view(np.uint32).fill(0xdeadbeef)
268  xi = xi[slices]
269  xi[...] = x
270  xi = xi.view(x.__class__)
271  assert_(np.all(xi == x))
272  yield xi, "stride_" + "_".join(["%+d" % j for j in repeats])
273 
274  # generate also zero strides if possible
275  if x.ndim >= 1 and x.shape[-1] == 1:
276  s = list(x.strides)
277  s[-1] = 0
278  xi = np.lib.stride_tricks.as_strided(x, strides=s)
279  yield xi, "stride_xxx_0"
280  if x.ndim >= 2 and x.shape[-2] == 1:
281  s = list(x.strides)
282  s[-2] = 0
283  xi = np.lib.stride_tricks.as_strided(x, strides=s)
284  yield xi, "stride_xxx_0_x"
285  if x.ndim >= 2 and x.shape[:-2] == (1, 1):
286  s = list(x.strides)
287  s[-1] = 0
288  s[-2] = 0
289  xi = np.lib.stride_tricks.as_strided(x, strides=s)
290  yield xi, "stride_xxx_0_0"
