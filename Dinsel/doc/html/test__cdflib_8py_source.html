58 class EndpointFilter(object):
59  def __init__(self, a, b, rtol, atol):
60  self.a = a
61  self.b = b
62  self.rtol = rtol
63  self.atol = atol
64 
65  def __call__(self, x):
66  mask1 = np.abs(x - self.a) < self.rtol*np.abs(self.a) + self.atol
67  mask2 = np.abs(x - self.b) < self.rtol*np.abs(self.b) + self.atol
68  return np.where(mask1 | mask2, False, True)

156 def _binomial_cdf(k, n, p):
157  k, n, p = mpmath.mpf(k), mpmath.mpf(n), mpmath.mpf(p)
158  if k <= 0:
159  return mpmath.mpf(0)
160  elif k >= n:
161  return mpmath.mpf(1)
162 
163  onemp = mpmath.fsub(1, p, exact=True)
164  return mpmath.betainc(n - k, k + 1, x2=onemp, regularized=True)

167 def _f_cdf(dfn, dfd, x):
168  if x < 0:
169  return mpmath.mpf(0)
170  dfn, dfd, x = mpmath.mpf(dfn), mpmath.mpf(dfd), mpmath.mpf(x)
171  ub = dfn*x/(dfn*x + dfd)
172  res = mpmath.betainc(dfn/2, dfd/2, x2=ub, regularized=True)
173  return res

176 def _student_t_cdf(df, t, dps=None):
177  if dps is None:
178  dps = mpmath.mp.dps
179  with mpmath.workdps(dps):
180  df, t = mpmath.mpf(df), mpmath.mpf(t)
181  fac = mpmath.hyp2f1(0.5, 0.5*(df + 1), 1.5, -t**2/df)
182  fac *= t*mpmath.gamma(0.5*(df + 1))
183  fac /= mpmath.sqrt(mpmath.pi*df)*mpmath.gamma(0.5*df)
184  return 0.5 + fac

187 def _noncentral_chi_pdf(t, df, nc):
188  res = mpmath.besseli(df/2 - 1, mpmath.sqrt(nc*t))
189  res *= mpmath.exp(-(t + nc)/2)*(t/nc)**(df/4 - 1/2)/2

194  if dps is None:
195  dps = mpmath.mp.dps
