78  def get_ancestors(self, node, level):
79  """return ancestor nodes of a class node"""
80  if level == 0:
81  return
82  for ancestor in node.ancestors(recurs=False):
83  if not self.show_node(ancestor):
84  continue
85  yield ancestor

87  def get_associated(self, klass_node, level):
88  """return associated nodes of a class node"""
89  if level == 0:
90  return
91  for ass_nodes in list(klass_node.instance_attrs_type.values()) + \
92  list(klass_node.locals_type.values()):
93  for ass_node in ass_nodes:
94  if isinstance(ass_node, astroid.Instance):
95  ass_node = ass_node._proxied
96  if not (isinstance(ass_node, astroid.ClassDef)
97  and self.show_node(ass_node)):
98  continue
99  yield ass_node

101  def extract_classes(self, klass_node, anc_level, ass_level):
102  """extract recursively classes related to klass_node"""
103  if self.classdiagram.has_node(klass_node) or not self.show_node(klass_node):
104  return
105  self.add_class(klass_node)
106 
107  for ancestor in self.get_ancestors(klass_node, anc_level):
108  self.extract_classes(ancestor, anc_level-1, ass_level)
109 
110  for ass_node in self.get_associated(klass_node, ass_level):
111  self.extract_classes(ass_node, anc_level, ass_level-1)

170 class ClassDiadefGenerator(DiaDefGenerator):
171  """generate a class diagram definition including all classes related to a
172  given class
173  """

178  def class_diagram(self, project, klass):
179  """return a class diagram definition for the given klass and its
180  related klasses
181  """
182 
183  self.classdiagram = ClassDiagram(klass, self.config.mode)
184  if len(project.modules) > 1:
185  module, klass = klass.rsplit('.', 1)
186  module = project.get_module(module)
187  else:
188  module = project.modules[0]
189  klass = klass.split('.')[-1]
190  klass = next(module.ilookup(klass))
191 
192  anc_level, ass_level = self._get_levels()
193  self.extract_classes(klass, anc_level, ass_level)
194  return self.classdiagram

207  def get_diadefs(self, project, linker):
208  """Get the diagrams configuration data
209 
210  :param project:The pyreverse project
211  :type project: pyreverse.utils.Project
212  :param linker: The linker
213  :type linker: pyreverse.inspector.Linker(IdGeneratorMixIn, LocalsVisitor)
214 
215  :returns: The list of diagram definitions
216  :rtype: list(:class:`pylint.pyreverse.diagrams.ClassDiagram`)
217  """
218 
219  # read and interpret diagram definitions (Diadefs)
220  diagrams = []
221  generator = ClassDiadefGenerator(linker, self)
222  for klass in self.config.classes:
223  diagrams.append(generator.class_diagram(project, klass))
224  if not diagrams:
