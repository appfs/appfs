125 def _format_marker(marker, first=True):
126  assert isinstance(marker, (list, tuple, string_types))
127 
128  # Sometimes we have a structure like [[...]] which is a single item list
129  # where the single item is itself it's own list. In that case we want skip
130  # the rest of this function so that we don't get extraneous () on the
131  # outside.
132  if (isinstance(marker, list) and len(marker) == 1 and
133  isinstance(marker[0], (list, tuple))):
134  return _format_marker(marker[0])
135 
136  if isinstance(marker, list):
137  inner = (_format_marker(m, first=False) for m in marker)
138  if first:
139  return " ".join(inner)
140  else:
141  return "(" + " ".join(inner) + ")"
142  elif isinstance(marker, tuple):
143  return '{0} {1} "{2}"'.format(*marker)
144  else:
145  return marker

160 def _eval_op(lhs, op, rhs):
161  try:
162  spec = Specifier("".join([op, rhs]))

191 def _evaluate_markers(markers, environment):
192  groups = [[]]
193 
194  for marker in markers:
195  assert isinstance(marker, (list, tuple, string_types))
196 
197  if isinstance(marker, list):
198  groups[-1].append(_evaluate_markers(marker, environment))
199  elif isinstance(marker, tuple):
200  lhs, op, rhs = marker
201 
202  if isinstance(lhs, Variable):
203  lhs_value = _get_env(environment, lhs.value)
204  rhs_value = rhs.value
205  else:
206  lhs_value = lhs.value
207  rhs_value = _get_env(environment, rhs.value)
208 
209  groups[-1].append(_eval_op(lhs_value, op, rhs_value))
210  else:
211  assert marker in ["and", "or"]
212  if marker == "or":
213  groups.append([])
214 
215  return any(all(item) for item in groups)
