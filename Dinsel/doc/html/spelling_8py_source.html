160  def _check_spelling(self, msgid, line, line_num):
161  for word, _ in self.tokenizer(line.strip()):
162  # Skip words from ignore list.
163  if word in self.ignore_list:
164  continue
165 
166  orig_word = word
167  word = word.lower()
168 
169  # Strip starting u' from unicode literals and r' from raw strings.
170  if (word.startswith("u'") or
171  word.startswith('u"') or
172  word.startswith("r'") or
173  word.startswith('r"')) and len(word) > 2:
174  word = word[2:]
175 
176  # If it is a known word, then continue.
177  try:
178  if self.spelling_dict.check(word):
179  continue
180  except enchant.errors.Error:
181  # this can only happen in docstrings, not comments
182  self.add_message('invalid-characters-in-docstring',
183  line=line_num, args=(word,))
184  continue
185 
186  # Store word to private dict or raise a message.
187  if self.config.spelling_store_unknown_words:
188  if word not in self.unknown_words:
189  self.private_dict_file.write("%s\n" % word)
190  self.unknown_words.add(word)
191  else:
192  # Present up to 4 suggestions.
193  # TODO: add support for customising this.
194  suggestions = self.spelling_dict.suggest(word)[:4]
195 
196  m = re.search(r"(\W|^)(%s)(\W|$)" % word, line.lower())
197  if m:
198  # Start position of second group in regex.
199  col = m.regs[2][0]
200  else:
201  col = line.lower().index(word)
202  indicator = (" " * col) + ("^" * len(word))
203 
204  self.add_message(msgid, line=line_num,
205  args=(orig_word, line,
206  indicator,
207  "'{0}'".format("' or '".join(suggestions))))
