80 class BroydenTridiagonal(object):
81  def __init__(self, n=100, mode='sparse'):
82  np.random.seed(0)
83 
84  self.n = n
85 
86  self.x0 = -np.ones(n)
87  self.lb = np.linspace(-2, -1.5, n)
88  self.ub = np.linspace(-0.8, 0.0, n)
89 
90  self.lb += 0.1 * np.random.randn(n)
91  self.ub += 0.1 * np.random.randn(n)
92 
93  self.x0 += 0.1 * np.random.randn(n)
94  self.x0 = make_strictly_feasible(self.x0, self.lb, self.ub)
95 
96  if mode == 'sparse':
97  self.sparsity = lil_matrix((n, n), dtype=int)
98  i = np.arange(n)
99  self.sparsity[i, i] = 1
100  i = np.arange(1, n)
101  self.sparsity[i, i - 1] = 1
102  i = np.arange(n - 1)
103  self.sparsity[i, i + 1] = 1
104 
105  self.jac = self._jac
106  elif mode == 'operator':
107  self.jac = lambda x: aslinearoperator(self._jac(x))
108  elif mode == 'dense':
109  self.sparsity = None
110  self.jac = lambda x: self._jac(x).toarray()
111  else:
112  assert_(False)

114  def fun(self, x):
115  f = (3 - x) * x + 1
116  f[1:] -= x[:-1]
117  f[:-1] -= 2 * x[1:]
118  return f

120  def _jac(self, x):
121  J = lil_matrix((self.n, self.n))
122  i = np.arange(self.n)
123  J[i, i] = 3 - 2 * x
124  i = np.arange(1, self.n)
125  J[i, i - 1] = -1
126  i = np.arange(self.n - 1)
127  J[i, i + 1] = -2
128  return J

131 class ExponentialFittingProblem(object):
132  """Provide data and function for exponential fitting in the form
133  y = a + exp(b * x) + noise."""
134 
135  def __init__(self, a, b, noise, n_outliers=1, x_range=(-1, 1),
136  n_points=11, random_seed=None):
137  np.random.seed(random_seed)
138  self.m = n_points
139  self.n = 2
140 
141  self.p0 = np.zeros(2)
142  self.x = np.linspace(x_range[0], x_range[1], n_points)
143 
144  self.y = a + np.exp(b * self.x)
145  self.y += noise * np.random.randn(self.m)
146 
147  outliers = np.random.randint(0, self.m, n_outliers)
148  self.y[outliers] += 50 * noise * np.random.rand(n_outliers)
149 
150  self.p_opt = np.array([a, b])
151 
152  def fun(self, p):
153  return p[0] + np.exp(p[1] * self.x) - self.y
154 
155  def jac(self, p):
156  J = np.empty((self.m, self.n))
157  J[:, 0] = 1
158  J[:, 1] = self.x * np.exp(p[1] * self.x)
159  return J

597  def test_jac(self):
598  # Test that res.jac.T.dot(res.jac) gives Gauss-Newton approximation
599  # of Hessian. This approximation is computed by doubly differentiating
600  # the cost function and dropping the part containing second derivative
601  # of f. For a scalar function it is computed as
602  # H = (rho' + 2 * rho'' * f**2) * f'**2, if the expression inside the
603  # brackets is less than EPS it is replaced by EPS. Here we check
604  # against the root of H.
605 
606  x = 2.0 # res.x will be this.
607  f = x**2 + 5 # res.fun will be this.
608 
609  res = least_squares(fun_trivial, x, jac_trivial, loss='linear',
610  max_nfev=1, method=self.method)
611  assert_equal(res.jac, 2 * x)
612 
613  # For `huber` loss the Jacobian correction is identically zero
614  # in outlier region, in such cases it is modified to be equal EPS**0.5.
615  res = least_squares(fun_trivial, x, jac_trivial, loss='huber',
616  max_nfev=1, method=self.method)
617  assert_equal(res.jac, 2 * x * EPS**0.5)
618 
619  # Now let's apply `loss_scale` to turn the residual into an inlier.
620  # The loss function becomes linear.
621  res = least_squares(fun_trivial, x, jac_trivial, loss='huber',
622  f_scale=10, max_nfev=1)
623  assert_equal(res.jac, 2 * x)
624 
625  # 'soft_l1' always gives a positive scaling.
626  res = least_squares(fun_trivial, x, jac_trivial, loss='soft_l1',
627  max_nfev=1, method=self.method)
628  assert_allclose(res.jac, 2 * x * (1 + f**2)**-0.75)
629 
630  # For 'cauchy' the correction term turns out to be negative, and it
631  # replaced by EPS**0.5.
632  res = least_squares(fun_trivial, x, jac_trivial, loss='cauchy',
633  max_nfev=1, method=self.method)
634  assert_allclose(res.jac, 2 * x * EPS**0.5)
635 
636  # Now use scaling to turn the residual to inlier.
637  res = least_squares(fun_trivial, x, jac_trivial, loss='cauchy',
638  f_scale=10, max_nfev=1, method=self.method)
639  fs = f / 10
640  assert_allclose(res.jac, 2 * x * (1 - fs**2)**0.5 / (1 + fs**2))
641 
642  # 'arctan' gives an outlier.
643  res = least_squares(fun_trivial, x, jac_trivial, loss='arctan',
644  max_nfev=1, method=self.method)
645  assert_allclose(res.jac, 2 * x * EPS**0.5)
