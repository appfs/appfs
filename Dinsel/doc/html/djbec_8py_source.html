224 def curve25519(n, base=9):
225  one = (base,1)
226  two = double(one)
227  # f(m) evaluates to a tuple
228  # containing the mth multiple and the
229  # (m+1)th multiple of base.
230  def f(m):
231  if m == 1: return (one, two)

234  return (add(pm, pm1, one), double(pm1))
