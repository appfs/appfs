197 """the assigned_stmts method is responsible to return the assigned statement
198 (e.g. not inferred) according to the assignment type.
199 
200 The `asspath` argument is used to record the lhs path of the original node.
201 For instance if we want assigned statements for 'c' in 'a, (b,c)', asspath
202 will be [1, 1] once arrived to the Assign node.
203 
204 The `context` argument is the current inference context which should be given
205 to any intermediary inference necessary.
206 """

208 def _resolve_looppart(parts, asspath, context):
209  """recursive function to resolve multiple assignments on loops"""
210  asspath = asspath[:]
211  index = asspath.pop(0)
212  for part in parts:
213  if part is util.Uninferable:
214  continue
215  # XXX handle __iter__ and log potentially detected errors
216  if not hasattr(part, 'itered'):
217  continue
218  try:
219  itered = part.itered()
220  except TypeError:
221  continue # XXX log error
222  for stmt in itered:
223  index_node = nodes.Const(index)
224  try:
225  assigned = stmt.getitem(index_node, context)
226  except (AttributeError,
227  exceptions.AstroidTypeError,
228  exceptions.AstroidIndexError):
229  continue
230  if not asspath:
231  # we achieved to resolved the assignment path,
232  # don't infer the last part
233  yield assigned
234  elif assigned is util.Uninferable:
235  break
236  else:
237  # we are not yet on the last part of the path
238  # search on each possibly inferred value
239  try:
240  for inferred in _resolve_looppart(assigned.infer(context),
241  asspath, context):
242  yield inferred
243  except exceptions.InferenceError:
244  break

293 def _arguments_infer_argname(self, name, context):
294  # arguments information may be missing, in which case we can't do anything
295  # more
296  if not (self.args or self.vararg or self.kwarg):
297  yield util.Uninferable
298  return
299  # first argument of instance/class method
300  if self.args and getattr(self.args[0], 'name', None) == name:
301  functype = self.parent.type
302  cls = self.parent.parent.scope()
303  is_metaclass = isinstance(cls, nodes.ClassDef) and cls.type == 'metaclass'
304  # If this is a metaclass, then the first argument will always
305  # be the class, not an instance.
306  if is_metaclass or functype == 'classmethod':
307  yield cls
308  return
309  if functype == 'method':
310  yield bases.Instance(self.parent.parent.frame())
311  return
312 
313  if context and context.callcontext:
314  call_site = arguments.CallSite(context.callcontext)
315  for value in call_site.infer_argument(self.parent, name, context):
316  yield value
317  return
318 
319  # TODO: just provide the type here, no need to have an empty Dict.
320  if name == self.vararg:
321  vararg = nodes.const_factory(())
322  vararg.parent = self
323  yield vararg
324  return
325  if name == self.kwarg:
326  kwarg = nodes.const_factory({})
327  kwarg.parent = self
328  yield kwarg
329  return
330  # if there is a default value, yield it. And then yield Uninferable to reflect
331  # we can't guess given argument value
332  try:
333  context = contextmod.copy_context(context)
334  for inferred in self.default_value(name).infer(context):
335  yield inferred
336  yield util.Uninferable
337  except exceptions.NoDefault:
338  yield util.Uninferable

426  try:
427  inferred = next(mgr.infer(context=context))
428  except exceptions.InferenceError:
429  return

431  # Check if it is decorated with contextlib.contextmanager.
432  func = inferred.parent
433  if not func.decorators:

445  # then a RuntimeError will be raised.

471 @decorators.raise_if_nothing_inferred
472 def with_assigned_stmts(self, node=None, context=None, asspath=None):
473  """Infer names and other nodes from a *with* statement.
474 
475  This enables only inference for name binding in a *with* statement.
476  For instance, in the following code, inferring `func` will return
477  the `ContextManager` class, not whatever ``__enter__`` returns.
478  We are doing this intentionally, because we consider that the context
479  manager result is whatever __enter__ returns and what it is binded
480  using the ``as`` keyword.
481 
482  class ContextManager(object):
483  def __enter__(self):
484  return 42
485  with ContextManager() as f:
486  pass
487  # ContextManager().infer() will return ContextManager
488  # f.infer() will return 42.
489 
490  Arguments:
491  self: nodes.With
492  node: The target of the assignment, `as (a, b)` in `with foo as (a, b)`.
493  context: TODO
494  asspath: TODO
495  """
496  mgr = next(mgr for (mgr, vars) in self.items if vars == node)
497  if asspath is None:
498  for result in _infer_context_manager(self, mgr, context):
499  yield result
500  else:
501  for result in _infer_context_manager(self, mgr, context):
502  # Walk the asspath and get the item at the final index.
503  obj = result
504  for index in asspath:
505  if not hasattr(obj, 'elts'):
506  raise exceptions.InferenceError(
507  'Wrong type ({targets!r}) for {node!r} assignment',
508  node=self, targets=node, assign_path=asspath,
509  context=context)
510  try:
511  obj = obj.elts[index]
512  except IndexError:
513  util.reraise(exceptions.InferenceError(
514  'Tried to infer a nonexistent target with index {index} '
515  'in {node!r}.', node=self, targets=node,
516  assign_path=asspath, context=context))
517  yield obj

519  # in raise_if_nothing_inferred.

527 def starred_assigned_stmts(self, node=None, context=None, asspath=None):
528  """
529  Arguments:
530  self: nodes.Starred
531  node: TODO
532  context: TODO
533  asspath: TODO
534  """
535  stmt = self.statement()
536  if not isinstance(stmt, (nodes.Assign, nodes.For)):
537  raise exceptions.InferenceError('Statement {stmt!r} enclosing {node!r} '
538  'must be an Assign or For node.',
539  node=self, stmt=stmt, unknown=node,
540  context=context)
541 
542  if isinstance(stmt, nodes.Assign):
543  value = stmt.value
544  lhs = stmt.targets[0]
545 
546  if sum(1 for node in lhs.nodes_of_class(nodes.Starred)) > 1:
547  raise exceptions.InferenceError('Too many starred arguments in the '
548  ' assignment targets {lhs!r}.',
549  node=self, targets=lhs,
550  unknown=node, context=context)
551 
552  if context is None:
553  context = contextmod.InferenceContext()
554  try:
555  rhs = next(value.infer(context))
556  except exceptions.InferenceError:
557  yield util.Uninferable
558  return
559  if rhs is util.Uninferable or not hasattr(rhs, 'elts'):
560  # Not interested in inferred values without elts.
561  yield util.Uninferable
562  return
563 
564  elts = collections.deque(rhs.elts[:])
565  if len(lhs.elts) > len(rhs.elts):
566  raise exceptions.InferenceError('More targets, {targets!r}, than '
567  'values to unpack, {values!r}.',
568  node=self, targets=lhs,
569  values=rhs, unknown=node,
570  context=context)
571 
572  # Unpack iteratively the values from the rhs of the assignment,
573  # until the find the starred node. What will remain will
574  # be the list of values which the Starred node will represent
575  # This is done in two steps, from left to right to remove
576  # anything before the starred node and from right to left
577  # to remove anything after the starred node.
578 
579  for index, left_node in enumerate(lhs.elts):
580  if not isinstance(left_node, nodes.Starred):
581  elts.popleft()
582  continue
583  lhs_elts = collections.deque(reversed(lhs.elts[index:]))
584  for right_node in lhs_elts:
585  if not isinstance(right_node, nodes.Starred):
586  elts.pop()
587  continue
588  # We're done
589  packed = nodes.List()
590  packed.elts = elts
591  packed.parent = self
592  yield packed
593  break
