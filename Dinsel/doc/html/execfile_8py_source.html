58  def find_module(modulename):
59  """Find the module named `modulename`.
60 
61  Returns the file path of the module, and the name of the enclosing
62  package.
63  """
64  openfile = None
65  glo, loc = globals(), locals()
66  try:
67  # Search for the module - inside its parent package, if any - using
68  # standard import mechanics.
69  if '.' in modulename:
70  packagename, name = modulename.rsplit('.', 1)
71  package = __import__(packagename, glo, loc, ['__path__'])
72  searchpath = package.__path__
73  else:
74  packagename, name = None, modulename
75  searchpath = None # "top-level search" in imp.find_module()
76  openfile, pathname, _ = imp.find_module(name, searchpath)
77 
78  # Complain if this is a magic non-file module.
79  if openfile is None and pathname is None:
80  raise NoSource(
81  "module does not live in a file: %r" % modulename
82  )
83 
84  # If `modulename` is actually a package, not a mere module, then we
85  # pretend to be Python 2.7 and try running its __main__.py script.
86  if openfile is None:
87  packagename = modulename
88  name = '__main__'
89  package = __import__(packagename, glo, loc, ['__path__'])
90  searchpath = package.__path__
91  openfile, pathname, _ = imp.find_module(name, searchpath)
92  except ImportError as err:
93  raise NoSource(str(err))
94  finally:
95  if openfile:
96  openfile.close()

116 def run_python_file(filename, args, package=None, modulename=None, path0=None):
117  """Run a Python file as if it were the main program on the command line.
118 
119  `filename` is the path to the file to execute, it need not be a .py file.
120  `args` is the argument array to present as sys.argv, including the first
121  element naming the file being executed. `package` is the name of the
122  enclosing package, if any.
123 
124  `modulename` is the name of the module the file was run as.
125 
126  `path0` is the value to put into sys.path[0]. If it's None, then this
127  function will decide on a value.
128 
129  """
130  if modulename is None and sys.version_info >= (3, 3):
131  modulename = '__main__'
132 
133  # Create a module to serve as __main__
134  old_main_mod = sys.modules['__main__']
135  main_mod = types.ModuleType('__main__')
136  sys.modules['__main__'] = main_mod
137  main_mod.__file__ = filename
138  if package:
139  main_mod.__package__ = package
140  if modulename:
141  main_mod.__loader__ = DummyLoader(modulename)
142 
143  main_mod.__builtins__ = BUILTINS
144 
145  # Set sys.argv properly.
146  old_argv = sys.argv
147  sys.argv = args
148 
149  if os.path.isdir(filename):
150  # Running a directory means running the __main__.py file in that
151  # directory.
152  my_path0 = filename
153 
154  for ext in [".py", ".pyc", ".pyo"]:
155  try_filename = os.path.join(filename, "__main__" + ext)
156  if os.path.exists(try_filename):
157  filename = try_filename
158  break
159  else:
160  raise NoSource("Can't find '__main__' module in '%s'" % filename)
161  else:
162  my_path0 = os.path.abspath(os.path.dirname(filename))
163 
164  # Set sys.path correctly.
165  old_path0 = sys.path[0]
166  sys.path[0] = path0 if path0 is not None else my_path0
167 
168  try:
169  try:
170  # Make a code object somehow.
171  if filename.endswith((".pyc", ".pyo")):
172  code = make_code_from_pyc(filename)
173  else:
174  code = make_code_from_py(filename)
175  except CoverageException:
176  raise
177  except Exception as exc:
178  msg = "Couldn't run {filename!r} as Python code: {exc.__class__.__name__}: {exc}"
179  raise CoverageException(msg.format(filename=filename, exc=exc))
180 
181  # Execute the code object.
182  try:
183  exec(code, main_mod.__dict__)
184  except SystemExit:
185  # The user called sys.exit(). Just pass it along to the upper
186  # layers, where it will be handled.
187  raise
188  except Exception:
189  # Something went wrong while executing the user code.
190  # Get the exc_info, and pack them into an exception that we can
191  # throw up to the outer loop. We peel one layer off the traceback
192  # so that the coverage.py code doesn't appear in the final printed
193  # traceback.
194  typ, err, tb = sys.exc_info()
195 
196  # PyPy3 weirdness. If I don't access __context__, then somehow it
197  # is non-None when the exception is reported at the upper layer,
198  # and a nested exception is shown to the user. This getattr fixes
199  # it somehow? https://bitbucket.org/pypy/pypy/issue/1903
200  getattr(err, '__context__', None)
201 
202  # Call the excepthook.
203  try:
204  if hasattr(err, "__traceback__"):
205  err.__traceback__ = err.__traceback__.tb_next
206  sys.excepthook(typ, err, tb.tb_next)
207  except SystemExit:
208  raise
209  except Exception:
210  # Getting the output right in the case of excepthook
211  # shenanigans is kind of involved.
212  sys.stderr.write("Error in sys.excepthook:\n")
213  typ2, err2, tb2 = sys.exc_info()
214  err2.__suppress_context__ = True
215  if hasattr(err2, "__traceback__"):
216  err2.__traceback__ = err2.__traceback__.tb_next
217  sys.__excepthook__(typ2, err2, tb2.tb_next)
218  sys.stderr.write("\nOriginal exception was:\n")
219  raise ExceptionDuringRun(typ, err, tb.tb_next)
220  else:
221  sys.exit(1)
222 
223  finally:
224  # Restore the old __main__, argv, and path.
225  sys.modules['__main__'] = old_main_mod
226  sys.argv = old_argv
227  sys.path[0] = old_path0

242 def make_code_from_pyc(filename):
243  """Get a code object from a .pyc file."""
244  try:
245  fpyc = open(filename, "rb")
246  except IOError:
247  raise NoCode("No file to run: '%s'" % filename)
248 
249  with fpyc:
250  # First four bytes are a version-specific magic number. It has to
251  # match or we won't run the file.
252  magic = fpyc.read(4)
253  if magic != PYC_MAGIC_NUMBER:
254  raise NoCode("Bad magic number in .pyc file")
255 
256  # Skip the junk in the header that we don't need.
257  fpyc.read(4) # Skip the moddate.
258  if sys.version_info >= (3, 3):
259  # 3.3 added another long to the header (size), skip it.
260  fpyc.read(4)
261 
262  # The rest of the file is the code object we want.
263  code = marshal.load(fpyc)
264 
265  return code
