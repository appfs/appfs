31 def _check_mode_str(mode):
32  # check type
33  if not isinstance(mode, six.string_types):
34  return False
35  # check syntax
36  modes = set(mode)
37  _mode = "rwatb+U"
38  creating = False
39  if six.PY3:
40  _mode += "x"
41  creating = "x" in modes
42  if modes - set(_mode) or len(mode) > len(modes):
43  return False
44  # check logic
45  reading = "r" in modes
46  writing = "w" in modes
47  appending = "a" in modes
48  text = "t" in modes
49  binary = "b" in modes
50  if "U" in modes:
51  if writing or appending or creating and six.PY3:
52  return False
53  reading = True
54  if not six.PY3:
55  binary = True
56  if text and binary:
57  return False
58  total = reading + writing + appending + (creating if six.PY3 else 0)
59  if total > 1:
60  return False
61  if not (reading or writing or appending or creating and six.PY3):
62  return False
63  # other 2.x constraints
64  if not six.PY3:
65  if "U" in mode:
66  mode = mode.replace("U", "")
67  if "r" not in mode:
68  mode = "r" + mode
69  return mode[0] in ("r", "w", "a", "U")
70  return True

181  def visit_call(self, node):
182  """Visit a CallFunc node."""
183  try:
184  for inferred in node.func.infer():
185  if inferred.root().name == OPEN_MODULE:
186  if getattr(node.func, 'name', None) in OPEN_FILES:
187  self._check_open_mode(node)
188  if inferred.root().name == UNITTEST_CASE:
189  self._check_redundant_assert(node, inferred)
190  self._check_deprecated_method(node, inferred)
191  except astroid.InferenceError:
192  return

212  def _check_deprecated_method(self, node, inferred):
213  py_vers = sys.version_info[0]
214 
215  if isinstance(node.func, astroid.Attribute):
216  func_name = node.func.attrname
217  elif isinstance(node.func, astroid.Name):
218  func_name = node.func.name
219  else:
220  # Not interested in other nodes.
221  return
222 
223  # Reject nodes which aren't of interest to us.
224  acceptable_nodes = (astroid.BoundMethod,
225  astroid.UnboundMethod,
226  astroid.FunctionDef)
227  if not isinstance(inferred, acceptable_nodes):
228  return
229 
230  qname = inferred.qname()
231  if qname in self.deprecated[0]:
232  self.add_message('deprecated-method', node=node,
233  args=(func_name, ))
234  else:
235  for since_vers, func_list in self.deprecated[py_vers].items():
236  if since_vers <= sys.version_info and qname in func_list:
237  self.add_message('deprecated-method', node=node,
238  args=(func_name, ))
239  break

249  def _check_datetime(self, node):
250  """ Check that a datetime was infered.

261  def _check_open_mode(self, node):
262  """Check that the mode argument of an open or file call is valid."""
263  try:
264  mode_arg = utils.get_argument_from_call(node, position=1,
265  keyword='mode')
266  except utils.NoSuchArgumentError:
267  return
268  if mode_arg:
269  mode_arg = utils.safe_infer(mode_arg)
270  if (isinstance(mode_arg, astroid.Const)

272  self.add_message('bad-open-mode', node=node,
