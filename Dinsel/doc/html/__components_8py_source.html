14 def cs_graph_components(x):
15  """
16  Determine connected components of a graph stored as a compressed
17  sparse row or column matrix.
18 
19  For speed reasons, the symmetry of the matrix x is not checked. A
20  nonzero at index `(i, j)` means that node `i` is connected to node
21  `j` by an edge. The number of rows/columns of the matrix thus
22  corresponds to the number of nodes in the graph.
23 
24  Parameters
25  -----------
26  x : array_like or sparse matrix, 2 dimensions
27  The adjacency matrix of the graph. Only the upper triangular part
28  is used.
29 
30  Returns
31  --------
32  n_comp : int
33  The number of connected components.
34  label : ndarray (ints, 1 dimension):
35  The label array of each connected component (-2 is used to
36  indicate empty rows in the matrix: 0 everywhere, including
37  diagonal). This array has the length of the number of nodes,
38  i.e. one label for each node of the graph. Nodes having the same
39  label belong to the same connected component.
40 
41  Notes
42  ------
43  The matrix is assumed to be symmetric and the upper triangular part
44  of the matrix is used. The matrix is converted to a CSR matrix unless
45  it is already a CSR.
46 
47  Examples
48  --------
49  >>> from scipy.sparse.csgraph import connected_components
50  >>> D = np.eye(4)
51  >>> D[0,1] = D[1,0] = 1
52  >>> cs_graph_components(D)
53  (3, array([0, 0, 1, 2]))
54  >>> from scipy.sparse import dok_matrix
55  >>> cs_graph_components(dok_matrix(D))
56  (3, array([0, 0, 1, 2]))
57 
58  """
59  try:
60  shape = x.shape
61  except AttributeError:
62  raise ValueError(_msg0)
63 
64  if not ((len(x.shape) == 2) and (x.shape[0] == x.shape[1])):
65  raise ValueError(_msg1 % x.shape)
66 
67  if isspmatrix(x):
68  x = x.tocsr()
69  else:
70  x = csr_matrix(x)
71 
72  label = np.empty((shape[0],), dtype=x.indptr.dtype)
73 
74  n_comp = _cs_graph_components(shape[0], x.indptr, x.indices, label)
75 
76  return n_comp, label
