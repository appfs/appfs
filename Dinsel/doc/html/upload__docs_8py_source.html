104  @staticmethod
105  def _build_part(item, sep_boundary):
106  key, values = item
107  title = '\nContent-Disposition: form-data; name="%s"' % key
108  # handle multiple entries for the same name
109  if not isinstance(values, list):
110  values = [values]

112  if isinstance(value, tuple):

117  yield sep_boundary
118  yield _encode(title)
119  yield b"\n\n"
120  yield value

122  yield b'\n' # write an extra newline (lurve Macs)

124  @classmethod
125  def _build_multipart(cls, data):
126  """
127  Build up the MIME payload for the POST data
128  """
129  boundary = b'--------------GHSKFJDLGDS7543FJKLFHRE75642756743254'
130  sep_boundary = b'\n--' + boundary
131  end_boundary = sep_boundary + b'--'
132  end_items = end_boundary, b"\n",
133  builder = functools.partial(
134  cls._build_part,
135  sep_boundary=sep_boundary,
136  )
137  part_groups = map(builder, data.items())
138  parts = itertools.chain.from_iterable(part_groups)
139  body_items = itertools.chain(parts, end_items)
140  content_type = 'multipart/form-data; boundary=%s' % boundary.decode('ascii')
141  return b''.join(body_items), content_type
