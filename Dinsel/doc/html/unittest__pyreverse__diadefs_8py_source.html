27 def _process_relations(relations):
28  """extract relation indices from a relation list"""
29  result = []
30  for rel_type, rels in six.iteritems(relations):
31  for rel in rels:
32  result.append( (rel_type, rel.from_object.title,
33  rel.to_object.title) )
34  result.sort()
35  return result
