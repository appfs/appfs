15 class FortranFile(object):
16  """
17  A file object for unformatted sequential files from Fortran code.
18 
19  Parameters
20  ----------
21  filename : file or str
22  Open file object or filename.
23  mode : {'r', 'w'}, optional
24  Read-write mode, default is 'r'.
25  header_dtype : dtype, optional
26  Data type of the header. Size and endiness must match the input/output file.
27 
28  Notes
29  -----
30  These files are broken up into records of unspecified types. The size of
31  each record is given at the start (although the size of this header is not
32  standard) and the data is written onto disk without any formatting. Fortran
33  compilers supporting the BACKSPACE statement will write a second copy of
34  the size to facilitate backwards seeking.
35 
36  This class only supports files written with both sizes for the record.
37  It also does not support the subrecords used in Intel and gfortran compilers
38  for records which are greater than 2GB with a 4-byte header.
39 
40  An example of an unformatted sequential file in Fortran would be written as::
41 
42  OPEN(1, FILE=myfilename, FORM='unformatted')
43 
44  WRITE(1) myvariable
45 
46  Since this is a non-standard file format, whose contents depend on the
47  compiler and the endianness of the machine, caution is advised. Files from
48  gfortran 4.8.0 and gfortran 4.1.2 on x86_64 are known to work.
49 
50  Consider using Fortran direct-access files or files from the newer Stream
51  I/O, which can be easily read by `numpy.fromfile`.
52 
53  Examples
54  --------
55  To create an unformatted sequential Fortran file:
56 
57  >>> from scipy.io import FortranFile
58  >>> f = FortranFile('test.unf', 'w')
59  >>> f.write_record(np.array([1,2,3,4,5], dtype=np.int32))
60  >>> f.write_record(np.linspace(0,1,20).reshape((5,-1)))
61  >>> f.close()
62 
63  To read this file:
64 
65  >>> from scipy.io import FortranFile
66  >>> f = FortranFile('test.unf', 'r')
67  >>> print(f.read_ints(dtype=np.int32))
68  [1 2 3 4 5]
69  >>> print(f.read_reals(dtype=float).reshape((5,-1)))
70  [[ 0. 0.05263158 0.10526316 0.15789474]
71  [ 0.21052632 0.26315789 0.31578947 0.36842105]
72  [ 0.42105263 0.47368421 0.52631579 0.57894737]
73  [ 0.63157895 0.68421053 0.73684211 0.78947368]
74  [ 0.84210526 0.89473684 0.94736842 1. ]]
75  >>> f.close()
76

78  def __init__(self, filename, mode='r', header_dtype=np.uint32):
79  if header_dtype is None:
80  raise ValueError('Must specify dtype')
81 
82  header_dtype = np.dtype(header_dtype)
83  if header_dtype.kind != 'u':

85

114  def read_record(self, dtype=None):
115  """
116  Reads a record of a given type from the file.
117 
118  Parameters
119  ----------
120  dtype : dtype, optional
121  Data type specifying the size and endiness of the data.
122 
123  Returns
124  -------
125  data : ndarray
126  A one-dimensional array object.
127 
128  Notes
129  -----
130  If the record contains a multi-dimensional array, calling reshape or
131  resize will restructure the array to the correct size.
132  Since Fortran multidimensional arrays are stored in column-major format,
133  this may have some non-intuitive consequences. If the variable was
134  declared as 'INTEGER var(5,4)', for example, var could be read with
135  'read_record(dtype=np.integer).reshape( (4,5) )' since Python uses
136  row-major ordering of indices.
137 
138  One can transpose to obtain the indices in the same order as in Fortran.
139 
140  For records that contain several variables or mixed types (as opposed
141  to single scalar or array types), it is possible to specify a dtype
142  with mixed types::
143 
144  record = f.read_record([('a', '<f4'), ('b', '<i4')])
145  record['a'] # access the variable 'a'
146 
147  and if any of the variables are arrays, the shape can be specified as
148  the third item in the relevant tuple::
149 
150  record = f.read_record([('a', '<f4'), ('b', '<i4', (3,3))])
151 
152  Numpy also supports a short syntax for this kind of type::
153 
154  record = f.read_record('<f4,(3,3)<i4')
155  record['f0'] # variables are called f0, f1, ...
156 
157 
158  See Also
159  --------
160  read_reals
161  read_ints
162 
163  """
164  if dtype is None:
165  raise ValueError('Must specify dtype')
166  dtype = np.dtype(dtype)
167 
168  firstSize = self._read_size()
169  if firstSize % dtype.itemsize != 0:
170  raise ValueError('Size obtained ({0}) is not a multiple of the '
171  'dtype given ({1}).'.format(firstSize, dtype.itemsize))
172 
173  data = np.fromfile(self._fp, dtype=dtype, count=firstSize//dtype.itemsize)
174  secondSize = self._read_size()
175  if firstSize != secondSize:
176  raise IOError('Sizes do not agree in the header and footer for '
177  'this record - check header dtype')
178  return data
