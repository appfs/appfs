151  def _check_simplifiable_if(self, node):
152  """Check if the given if node can be simplified.
153 
154  The if statement can be reduced to a boolean expression
155  in some cases. For instance, if there are two branches
156  and both of them return a boolean value that depends on
157  the result of the statement's test, then this can be reduced
158  to `bool(test)` without losing any functionality.
159  """
160 
161  if self._is_actual_elif(node):
162  # Not interested in if statements with multiple branches.
163  return
164  if len(node.orelse) != 1 or len(node.body) != 1:
165  return
166 
167  # Check if both branches can be reduced.
168  first_branch = node.body[0]
169  else_branch = node.orelse[0]
170  if isinstance(first_branch, astroid.Return):
171  if not isinstance(else_branch, astroid.Return):
172  return
173  first_branch_is_bool = self._is_bool_const(first_branch)
174  else_branch_is_bool = self._is_bool_const(else_branch)
175  reduced_to = "'return bool(test)'"
176  elif isinstance(first_branch, astroid.Assign):
177  if not isinstance(else_branch, astroid.Assign):
178  return
179  first_branch_is_bool = self._is_bool_const(first_branch)
180  else_branch_is_bool = self._is_bool_const(else_branch)
181  reduced_to = "'var = bool(test)'"
182  else:
183  return
184 
185  if not first_branch_is_bool or not else_branch_is_bool:
186  return
187  if not first_branch.value.value:
188  # This is a case that can't be easily simplified and
189  # if it can be simplified, it will usually result in a
190  # code that's harder to understand and comprehend.
191  # Let's take for instance `arg and arg <= 3`. This could theoretically be
192  # reduced to `not arg or arg > 3`, but the net result is that now the
193  # condition is harder to understand, because it requires understanding of
194  # an extra clause:
195  # * first, there is the negation of truthness with `not arg`
196  # * the second clause is `arg > 3`, which occurs when arg has a
197  # a truth value, but it implies that `arg > 3` is equivalent
198  # with `arg and arg > 3`, which means that the user must
199  # think about this assumption when evaluating `arg > 3`.
200  # The original form is easier to grasp.
201  return
202 
203  self.add_message('simplifiable-if-statement', node=node,
204  args=(reduced_to,))

314  def _check_nested_blocks(self, node):
315  """Update and check the number of nested blocks
316  """
317  # only check block levels inside functions or methods
318  if not isinstance(node.scope(), astroid.FunctionDef):
319  return
320  # messages are triggered on leaving the nested block. Here we save the
321  # stack in case the current node isn't nested in the previous one
322  nested_blocks = self._nested_blocks[:]
323  if node.parent == node.scope():
324  self._nested_blocks = [node]
325  else:
326  # go through ancestors from the most nested to the less
327  for ancestor_node in reversed(self._nested_blocks):
328  if ancestor_node == node.parent:
329  break
330  self._nested_blocks.pop()
331  # if the node is a elif, this should not be another nesting level
332  if isinstance(node, astroid.If) and self._elifs[self._if_counter]:
333  if self._nested_blocks:
334  self._nested_blocks.pop()
335  self._nested_blocks.append(node)
336 
337  # send message only once per group of nested blocks
338  if len(nested_blocks) > len(self._nested_blocks):
339  self._emit_nested_blocks_message_if_needed(nested_blocks)
