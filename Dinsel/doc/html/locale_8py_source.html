122 def _grouping_intervals(grouping):
123  last_interval = None

138 def _group(s, monetary=False):
139  conv = localeconv()
140  thousands_sep = conv[monetary and 'mon_thousands_sep' or 'thousands_sep']
141  grouping = conv[monetary and 'mon_grouping' or 'grouping']
142  if not grouping:
143  return (s, 0)
144  if s[-1] == ' ':
145  stripped = s.rstrip()
146  right_spaces = s[len(stripped):]
147  s = stripped
148  else:
149  right_spaces = ''
150  left_spaces = ''
151  groups = []
152  for interval in _grouping_intervals(grouping):
153  if not s or s[-1] not in "0123456789":
154  # only non-digit characters remain (sign, spaces)
155  left_spaces = s
156  s = ''
157  break
158  groups.append(s[-interval:])
159  s = s[:-interval]
160  if s:
161  groups.append(s)
162  groups.reverse()
163  return (
164  left_spaces + thousands_sep.join(groups) + right_spaces,
165  len(thousands_sep) * (len(groups) - 1)

256  in the current locale."""

375 def normalize(localename):
376 
377  """ Returns a normalized locale code for the given locale
378  name.
379 
380  The returned locale code is formatted for use with
381  setlocale().
382 
383  If normalization fails, the original name is returned
384  unchanged.
385 
386  If the given encoding is not known, the function defaults to
387  the default encoding for the locale code just like setlocale()
388  does.
389 
390  """
391  # Normalize the locale name and extract the encoding and modifier
392  code = localename.lower()
393  if ':' in code:
394  # ':' is sometimes used as encoding delimiter.
395  code = code.replace(':', '.')
396  if '@' in code:
397  code, modifier = code.split('@', 1)
398  else:
399  modifier = ''
400  if '.' in code:
401  langname, encoding = code.split('.')[:2]
402  else:
403  langname = code
404  encoding = ''
405 
406  # First lookup: fullname (possibly with encoding and modifier)
407  lang_enc = langname
408  if encoding:
409  norm_encoding = encoding.replace('-', '')
410  norm_encoding = norm_encoding.replace('_', '')
411  lang_enc += '.' + norm_encoding
412  lookup_name = lang_enc
413  if modifier:
414  lookup_name += '@' + modifier
415  code = locale_alias.get(lookup_name, None)
416  if code is not None:
417  return code
418  #print('first lookup failed')
419 
420  if modifier:
421  # Second try: fullname without modifier (possibly with encoding)
422  code = locale_alias.get(lang_enc, None)
423  if code is not None:
424  #print('lookup without modifier succeeded')
425  if '@' not in code:
426  return _append_modifier(code, modifier)
427  if code.split('@', 1)[1].lower() == modifier:
428  return code
429  #print('second lookup failed')
430 
431  if encoding:
432  # Third try: langname (without encoding, possibly with modifier)
433  lookup_name = langname
434  if modifier:
435  lookup_name += '@' + modifier
436  code = locale_alias.get(lookup_name, None)
437  if code is not None:
438  #print('lookup without encoding succeeded')
439  if '@' not in code:
440  return _replace_encoding(code, encoding)
441  code, modifier = code.split('@', 1)
442  return _replace_encoding(code, encoding) + '@' + modifier
443 
444  if modifier:
445  # Fourth try: langname (without encoding and modifier)
446  code = locale_alias.get(langname, None)
447  if code is not None:
448  #print('lookup without modifier and encoding succeeded')
449  if '@' not in code:
450  code = _replace_encoding(code, encoding)
451  return _append_modifier(code, modifier)
452  code, defmod = code.split('@', 1)
453  if defmod.lower() == modifier:
454  return _replace_encoding(code, encoding) + '@' + defmod
455 
456  return localename

508 def getdefaultlocale(envvars=('LC_ALL', 'LC_CTYPE', 'LANG', 'LANGUAGE')):
509 
510  """ Tries to determine the default locale settings and returns
511  them as tuple (language code, encoding).
512 
513  According to POSIX, a program which has not called
514  setlocale(LC_ALL, "") runs using the portable 'C' locale.
515  Calling setlocale(LC_ALL, "") lets it use the default locale as
516  defined by the LANG variable. Since we don't want to interfere
517  with the current locale setting we thus emulate the behavior
518  in the way described above.
519 
520  To maintain compatibility with other platforms, not only the
521  LANG variable is tested, but a list of variables given as
522  envvars parameter. The first found to be defined will be
523  used. envvars defaults to the search path used in GNU gettext;
524  it must always contain the variable name 'LANG'.
525 
526  Except for the code 'C', the language code corresponds to RFC
527  1766. code and encoding can be None in case the values cannot
528  be determined.
529 
530  """
531 
532  try:
533  # check if it's supported by the _locale module
534  import _locale
535  code, encoding = _locale._getdefaultlocale()
536  except (ImportError, AttributeError):
537  pass
538  else:
539  # make sure the code/encoding values are valid
540  if sys.platform == "win32" and code and code[:2] == "0x":
541  # map windows language identifier to language name
542  code = windows_locale.get(int(code, 0))
543  # ...add other platform-specific processing here, if
544  # necessary...
545  return code, encoding
546 
547  # fall back on POSIX behaviour
548  import os
549  lookup = os.environ.get
550  for variable in envvars:
551  localename = lookup(variable,None)
552  if localename:
553  if variable == 'LANGUAGE':
