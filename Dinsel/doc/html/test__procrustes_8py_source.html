75 def test_orthogonal_procrustes():
76  np.random.seed(1234)
77  for m, n in ((6, 4), (4, 4), (4, 6)):
78  # Sample a random target matrix.
79  B = np.random.randn(m, n)
80  # Sample a random orthogonal matrix
81  # by computing eigh of a sampled symmetric matrix.
82  X = np.random.randn(n, n)
83  w, V = eigh(X.T + X)
84  assert_allclose(inv(V), V.T)
85  # Compute a matrix with a known orthogonal transformation that gives B.
86  A = np.dot(B, V.T)
87  # Check that an orthogonal transformation from A to B can be recovered.
88  R, s = orthogonal_procrustes(A, B)
89  assert_allclose(inv(R), R.T)
90  assert_allclose(A.dot(R), B)
91  # Create a perturbed input matrix.
92  A_perturbed = A + 1e-2 * np.random.randn(m, n)
93  # Check that the orthogonal procrustes function can find an orthogonal
94  # transformation that is better than the orthogonal transformation
95  # computed from the original input matrix.
96  R_prime, s = orthogonal_procrustes(A_perturbed, B)
97  assert_allclose(inv(R_prime), R_prime.T)
98  # Compute the naive and optimal transformations of the perturbed input.
99  naive_approx = A_perturbed.dot(R)
100  optim_approx = A_perturbed.dot(R_prime)
101  # Compute the Frobenius norm errors of the matrix approximations.
102  naive_approx_error = norm(naive_approx - B, ord='fro')
103  optim_approx_error = norm(optim_approx - B, ord='fro')
104  # Check that the orthogonal Procrustes approximation is better.
105  assert_array_less(optim_approx_error, naive_approx_error)
