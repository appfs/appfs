232 def types_compatible(var1, var2):
233  """Check if types are same or compatible.
234 
235  0-D numpy scalars are compatible with bare python scalars.
236  """
237  type1 = type(var1)
238  type2 = type(var2)
239  if type1 is type2:
240  return True
241  if type1 is np.ndarray and var1.shape == ():
242  return type(var1.item()) is type2
243  if type2 is np.ndarray and var2.shape == ():
244  return type(var2.item()) is type1
245  return False

248 def _check_level(label, expected, actual):
249  """ Check one level of a potentially nested array """
250  if SP.issparse(expected): # allow different types of sparse matrices
251  assert_(SP.issparse(actual))
252  assert_array_almost_equal(actual.todense(),
253  expected.todense(),
254  err_msg=label,
255  decimal=5)
256  return
257  # Check types are as expected
258  assert_(types_compatible(expected, actual),
259  "Expected type %s, got %s at %s" %
260  (type(expected), type(actual), label))
261  # A field in a record array may not be an ndarray
262  # A scalar from a record array will be type np.void
263  if not isinstance(expected,
264  (np.void, np.ndarray, MatlabObject)):
265  assert_equal(expected, actual)
266  return
267  # This is an ndarray-like thing
268  assert_(expected.shape == actual.shape,
269  msg='Expected shape %s, got %s at %s' % (expected.shape,
270  actual.shape,
271  label))
272  ex_dtype = expected.dtype
273  if ex_dtype.hasobject: # array of objects
274  if isinstance(expected, MatlabObject):
275  assert_equal(expected.classname, actual.classname)
276  for i, ev in enumerate(expected):
277  level_label = "%s, [%d], " % (label, i)
278  _check_level(level_label, ev, actual[i])
279  return
280  if ex_dtype.fields: # probably recarray
281  for fn in ex_dtype.fields:
282  level_label = "%s, field %s, " % (label, fn)
283  _check_level(level_label,
284  expected[fn], actual[fn])
285  return
286  if ex_dtype.type in (text_type, # string or bool
287  np.unicode_,
288  np.bool_):
289  assert_equal(actual, expected, err_msg=label)
290  return
291  # Something numeric
292  assert_array_almost_equal(actual, expected, err_msg=label, decimal=5)
