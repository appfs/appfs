13 def linear_sum_assignment(cost_matrix):
14  """Solve the linear sum assignment problem.
15 
16  The linear sum assignment problem is also known as minimum weight matching
17  in bipartite graphs. A problem instance is described by a matrix C, where
18  each C[i,j] is the cost of matching vertex i of the first partite set
19  (a "worker") and vertex j of the second set (a "job"). The goal is to find
20  a complete assignment of workers to jobs of minimal cost.
21 
22  Formally, let X be a boolean matrix where :math:`X[i,j] = 1` iff row i is
23  assigned to column j. Then the optimal assignment has cost
24 
25  .. math::
26  \\min \\sum_i \\sum_j C_{i,j} X_{i,j}
27 
28  s.t. each row is assignment to at most one column, and each column to at
29  most one row.
30 
31  This function can also solve a generalization of the classic assignment
32  problem where the cost matrix is rectangular. If it has more rows than
33  columns, then not every row needs to be assigned to a column, and vice
34  versa.
35 
36  The method used is the Hungarian algorithm, also known as the Munkres or
37  Kuhn-Munkres algorithm.
38 
39  Parameters
40  ----------
41  cost_matrix : array
42  The cost matrix of the bipartite graph.
43 
44  Returns
45  -------
46  row_ind, col_ind : array
47  An array of row indices and one of corresponding column indices giving
48  the optimal assignment. The cost of the assignment can be computed
49  as ``cost_matrix[row_ind, col_ind].sum()``. The row indices will be
50  sorted; in the case of a square cost matrix they will be equal to
51  ``numpy.arange(cost_matrix.shape[0])``.
52 
53  Notes
54  -----
55  .. versionadded:: 0.17.0
56 
57  Examples
58  --------
59  >>> cost = np.array([[4, 1, 3], [2, 0, 5], [3, 2, 2]])
60  >>> from scipy.optimize import linear_sum_assignment
61  >>> row_ind, col_ind = linear_sum_assignment(cost)
62  >>> col_ind
63  array([1, 0, 2])
64  >>> cost[row_ind, col_ind].sum()
65  5
66 
67  References
68  ----------
69  1. http://csclab.murraystate.edu/bob.pilgrim/445/munkres.html
70 
71  2. Harold W. Kuhn. The Hungarian Method for the assignment problem.
72  *Naval Research Logistics Quarterly*, 2:83-97, 1955.
73 
74  3. Harold W. Kuhn. Variants of the Hungarian method for assignment
75  problems. *Naval Research Logistics Quarterly*, 3: 253-258, 1956.
76 
77  4. Munkres, J. Algorithms for the Assignment and Transportation Problems.
78  *J. SIAM*, 5(1):32-38, March, 1957.
79 
80  5. https://en.wikipedia.org/wiki/Hungarian_algorithm
81  """
82  cost_matrix = np.asarray(cost_matrix)
83  if len(cost_matrix.shape) != 2:
84  raise ValueError("expected a matrix (2-d array), got a %r array"
85  % (cost_matrix.shape,))
86 
87  # The algorithm expects more columns than rows in the cost matrix.
88  if cost_matrix.shape[1] < cost_matrix.shape[0]:
89  cost_matrix = cost_matrix.T
90  transposed = True
91  else:
92  transposed = False
93 
94  state = _Hungary(cost_matrix)
95 
96  # No need to bother with assignments if one of the dimensions
97  # of the cost matrix is zero-length.
98  step = None if 0 in cost_matrix.shape else _step1
99 
100  while step is not None:
101  step = step(state)
102 
103  if transposed:
104  marked = state.marked.T
105  else:
106  marked = state.marked
107  return np.where(marked == 1)

119  def __init__(self, cost_matrix):

136 # Individual steps of the algorithm follow, as a state machine: they return
137 # the next step to be taken (function to be called), if any.

139 def _step1(state):
140  """Steps 1 and 2 in the Wikipedia page."""
141 
142  # Step 1: For each row of the matrix, find the smallest element and
143  # subtract it from every element in its row.
144  state.C -= state.C.min(axis=1)[:, np.newaxis]
145  # Step 2: Find a zero (Z) in the resulting matrix. If there is no
146  # starred zero in its row or column, star Z. Repeat for each element
147  # in the matrix.
148  for i, j in zip(*np.where(state.C == 0)):
149  if state.col_uncovered[j] and state.row_uncovered[i]:
150  state.marked[i, j] = 1
151  state.col_uncovered[j] = False
152  state.row_uncovered[i] = False
153 
154  state._clear_covers()
155  return _step3

158 def _step3(state):
159  """
160  Cover each column containing a starred zero. If n columns are covered,
161  the starred zeros describe a complete set of unique assignments.
162  In this case, Go to DONE, otherwise, Go to Step 4.
163  """
164  marked = (state.marked == 1)
165  state.col_uncovered[np.any(marked, axis=0)] = False
166 
167  if marked.sum() < state.C.shape[0]:
168  return _step4

171 def _step4(state):
172  """
173  Find a noncovered zero and prime it. If there is no starred zero
174  in the row containing this primed zero, Go to Step 5. Otherwise,
175  cover this row and uncover the column containing the starred
176  zero. Continue in this manner until there are no uncovered zeros
177  left. Save the smallest uncovered value and Go to Step 6.
178  """
179  # We convert to int as numpy operations are faster on int
180  C = (state.C == 0).astype(int)
181  covered_C = C * state.row_uncovered[:, np.newaxis]
182  covered_C *= np.asarray(state.col_uncovered, dtype=int)
183  n = state.C.shape[0]
184  m = state.C.shape[1]
185 
186  while True:
187  # Find an uncovered zero
188  row, col = np.unravel_index(np.argmax(covered_C), (n, m))
189  if covered_C[row, col] == 0:
190  return _step6
191  else:
192  state.marked[row, col] = 2
193  # Find the first starred element in the row
194  star_col = np.argmax(state.marked[row] == 1)
195  if state.marked[row, star_col] != 1:
196  # Could not find one
197  state.Z0_r = row
198  state.Z0_c = col
199  return _step5
200  else:
201  col = star_col
202  state.row_uncovered[row] = False
203  state.col_uncovered[col] = True
204  covered_C[:, col] = C[:, col] * (
205  np.asarray(state.row_uncovered, dtype=int))
206  covered_C[row] = 0

209 def _step5(state):
210  """
211  Construct a series of alternating primed and starred zeros as follows.
212  Let Z0 represent the uncovered primed zero found in Step 4.
213  Let Z1 denote the starred zero in the column of Z0 (if any).
214  Let Z2 denote the primed zero in the row of Z1 (there will always be one).
215  Continue until the series terminates at a primed zero that has no starred
216  zero in its column. Unstar each starred zero of the series, star each
217  primed zero of the series, erase all primes and uncover every line in the
218  matrix. Return to Step 3
219  """
220  count = 0
221  path = state.path
222  path[count, 0] = state.Z0_r
223  path[count, 1] = state.Z0_c
224 
225  while True:
226  # Find the first starred element in the col defined by
227  # the path.
228  row = np.argmax(state.marked[:, path[count, 1]] == 1)
229  if state.marked[row, path[count, 1]] != 1:
230  # Could not find one
231  break
232  else:
233  count += 1
234  path[count, 0] = row
235  path[count, 1] = path[count - 1, 1]
236 
237  # Find the first prime element in the row defined by the
238  # first path step
239  col = np.argmax(state.marked[path[count, 0]] == 2)
240  if state.marked[row, col] != 2:
241  col = -1
242  count += 1
243  path[count, 0] = path[count - 1, 0]
244  path[count, 1] = col
245 
246  # Convert paths
247  for i in range(count + 1):
248  if state.marked[path[i, 0], path[i, 1]] == 1:
249  state.marked[path[i, 0], path[i, 1]] = 0
250  else:
251  state.marked[path[i, 0], path[i, 1]] = 1
252 
253  state._clear_covers()
254  # Erase all prime markings
255  state.marked[state.marked == 2] = 0
256  return _step3

259 def _step6(state):
260  """
261  Add the value found in Step 4 to every element of each covered row,
262  and subtract it from every element of each uncovered column.
263  Return to Step 4 without altering any stars, primes, or covered lines.
264  """
265  # the smallest uncovered value in the matrix
266  if np.any(state.row_uncovered) and np.any(state.col_uncovered):
267  minval = np.min(state.C[state.row_uncovered], axis=0)
268  minval = np.min(minval[state.col_uncovered])
269  state.C[~state.row_uncovered] += minval
270  state.C[:, state.col_uncovered] -= minval
271  return _step4
